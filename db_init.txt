CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

DROP TABLE logos;
CREATE TABLE IF NOT EXISTS logos(
  logo_id uuid DEFAULT uuid_generate_v4(),
  customer VARCHAR(50) NOT NULL,
  logo VARCHAR(50) NOT NULL,
  colors VARCHAR(50)[],
  text VARCHAR(50)[],
  link VARCHAR(300),
  info VARCHAR(1000),
  img BYTEA
);
